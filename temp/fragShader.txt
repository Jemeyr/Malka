#version 130

in vec3 fragNorm;
in vec2 fragTexCoord;

uniform vec3 color;

uniform sampler2D modelTexture;

void main(){
	float intensity = fragNorm.z + fragTexCoord.x;
	vec4 col = texture(modelTexture, fragTexCoord);
	
	intensity = intensity;
	
	gl_FragColor = intensity * vec4( color, 1.0 ) * col;
}
